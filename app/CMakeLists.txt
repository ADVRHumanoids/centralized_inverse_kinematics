find_package(YARP REQUIRED)

# pick up yarp's cmake scripts
list(APPEND CMAKE_MODULE_PATH ${YARP_MODULE_PATH})
include(YarpInstallationHelpers)	

file(GLOB conf conf/*.ini)

# directory name matches what is provided as "default context" in code
yarp_install(FILES ${conf} DESTINATION ${YARP_CONTEXTS_INSTALL_DIR}/centralized_inverse_kinematics)

add_custom_target(  copy_conf_files_centralized_inverse_kinematics ALL
		    COMMAND ${CMAKE_COMMAND} -E copy_directory
		    "${PROJECT_SOURCE_DIR}/app/conf"
		    "${CMAKE_BINARY_DIR}/"
		    )
